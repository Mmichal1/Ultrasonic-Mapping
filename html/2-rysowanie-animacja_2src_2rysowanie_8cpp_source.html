58 void OknoZRysunkiem::paintEvent( QPaintEvent * )
59 {
60  QPainter Rysownik(this);
61 
62  double ProporcjeWS_Obraz = _Niebo.height()/_Niebo.width();
63  double ProporcjeWS_Okno = height()/width();
64  double Skala;
65 
66  if (ProporcjeWS_Obraz > ProporcjeWS_Okno) {
67  Skala = height()/static_cast<double>(_Niebo.height());
68  } else {
69  Skala = width()/static_cast<double>(_Niebo.width());
70  }
71  Rysownik.scale(Skala,Skala);
72  Rysownik.drawImage(0,0,_Niebo);
73  Rysownik.drawImage(_Chmurka1_x,50,_Chmurka1);
74  Rysownik.drawImage(_Chmurka2_x,150,_Chmurka2);
