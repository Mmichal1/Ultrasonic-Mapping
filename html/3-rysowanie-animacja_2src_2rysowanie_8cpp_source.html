123 void OknoZRysunkiem::paintEvent( QPaintEvent * )
124 {
125  QPainter Rysownik(this);
126  Rysownik.setBackground(Qt::white);
127 
128  double ProporcjeWS_Obraz = _Niebo.height()/_Niebo.width();
129  double ProporcjeWS_Okno = height()/width();
130  double Skala;
131 
132  if (ProporcjeWS_Obraz > ProporcjeWS_Okno) {
133  Skala = height()/static_cast<double>(_Niebo.height());
134  } else {
135  Skala = width()/static_cast<double>(_Niebo.width());
136  }
137  Rysownik.scale(Skala,Skala);
138  Rysownik.drawImage(0,0,_Niebo);
139  Rysownik.drawImage(_Chmurka1_x,50,_Chmurka1);
140  Rysownik.drawImage(_Chmurka2_x,150,_Chmurka2);
