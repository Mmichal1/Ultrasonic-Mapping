152 void MainWindow::connectToPort(const QString &portName) {
153 
154  if (serial.isOpen()) {
155  serial.close();
156  }
157 
158  // Check if the selected port is still available
159  bool portExists = false;
160  foreach (const QSerialPortInfo &portInfo, QSerialPortInfo::availablePorts()) {
161  if (portInfo.portName() == portName) {
162  ui->connectionLabel->setPixmap(*connectionBadPixmap);
163  portExists = true;
164  break;
165  }
166  }
167 
168  if (portExists) {
169  serial.setPortName(portName);
170  if (serial.open(QIODevice::ReadWrite)) {
171  bar->showMessage(tr("Serial port opened successfully"), 2000);
172  stopStartButton->setText("Stop");
173  } else {
174  ui->connectionLabel->setPixmap(*connectionBadPixmap);
175  QByteArray ba = serial.errorString().toLocal8Bit();
176  bar->showMessage(tr(ba.data()), 2000);
177  }
178  } else {
179  ui->connectionLabel->setPixmap(*connectionBadPixmap);
